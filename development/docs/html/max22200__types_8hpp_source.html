224inline uint32_t getChopFreqKhz(bool master_clock_80khz, ChopFreq cf) {
225 switch (cf) {
226 case ChopFreq::FMAIN_DIV4: return master_clock_80khz ? 20u : 25u;
227 case ChopFreq::FMAIN_DIV3: return master_clock_80khz ? 26u : 33u;
228 case ChopFreq::FMAIN_DIV2: return master_clock_80khz ? 40u : 50u;
229 case ChopFreq::FMAIN: return master_clock_80khz ? 80u : 100u;
230 default: return 25u;
231 }

240inline uint8_t currentMaToRaw(uint32_t full_scale_current_ma, uint32_t ma) {
241 if (full_scale_current_ma == 0) return 0;
242 if (ma >= full_scale_current_ma) return 127;
243 uint32_t raw = (ma * 127u + full_scale_current_ma / 2u) / full_scale_current_ma;
244 return raw > 127u ? 127u : static_cast<uint8_t>(raw);
245}

254inline uint8_t hitTimeMsToRaw(float ms, bool master_clock_80khz, ChopFreq cf) {
255 uint32_t fchop_khz = getChopFreqKhz(master_clock_80khz, cf);
256 if (ms < 0.0f || ms >= 1000000.0f) return 255;
257 if (ms == 0.0f) return 0;
258 float fchop_hz = static_cast<float>(fchop_khz) * 1000.0f;
259 uint32_t raw = static_cast<uint32_t>((ms / 1000.0f) * fchop_hz / 40.0f + 0.5f);
260 if (raw > 254u) return 255;
261 if (raw == 0u) return 1;
262 return static_cast<uint8_t>(raw);
263}

410 uint32_t toRegister() const {
411 // Convert hit_setpoint to raw 7-bit
412 uint8_t hit_raw;
413 if (drive_mode == DriveMode::CDR) {
414 // CDR: hit_setpoint is in mA
415 if (full_scale_current_ma == 0) {
416 hit_raw = 0; // Invalid IFS
417 } else if (hit_setpoint >= static_cast<float>(full_scale_current_ma)) {
418 hit_raw = 127;
419 } else {
420 uint32_t ma = static_cast<uint32_t>(hit_setpoint + 0.5f);
421 hit_raw = currentMaToRaw(full_scale_current_ma, ma);
422 }
423 } else {
424 // VDR: hit_setpoint is duty percent (0-100)
425 if (hit_setpoint <= 0.0f) {
426 hit_raw = 0;
427 } else if (hit_setpoint >= 100.0f) {
428 hit_raw = 127;
429 } else {
430 uint32_t r = static_cast<uint32_t>((hit_setpoint / 100.0f) * 127.0f + 0.5f);
431 hit_raw = (r > 127u) ? 127u : static_cast<uint8_t>(r);
432 }
433 }
434
435 // Convert hold_setpoint to raw 7-bit
436 uint8_t hold_raw;
437 if (drive_mode == DriveMode::CDR) {
438 // CDR: hold_setpoint is in mA
439 if (full_scale_current_ma == 0) {
440 hold_raw = 0;
441 } else if (hold_setpoint >= static_cast<float>(full_scale_current_ma)) {
442 hold_raw = 127;
443 } else {
444 uint32_t ma = static_cast<uint32_t>(hold_setpoint + 0.5f);
445 hold_raw = currentMaToRaw(full_scale_current_ma, ma);
446 }
447 } else {
448 // VDR: hold_setpoint is duty percent (0-100)
449 if (hold_setpoint <= 0.0f) {
450 hold_raw = 0;
451 } else if (hold_setpoint >= 100.0f) {
452 hold_raw = 127;
453 } else {
454 uint32_t r = static_cast<uint32_t>((hold_setpoint / 100.0f) * 127.0f + 0.5f);
455 hold_raw = (r > 127u) ? 127u : static_cast<uint8_t>(r);
456 }
457 }
458
459 // Convert hit_time_ms to raw 8-bit
460 uint8_t hit_time_raw = hitTimeMsToRaw(hit_time_ms, master_clock_80khz, chop_freq);
461
462 // Build register value
463 uint32_t val = 0;
464 if (half_full_scale) val |= CfgChReg::HFS_BIT;
465 val |= (static_cast<uint32_t>(hold_raw & 0x7F) << CfgChReg::HOLD_SHIFT);
466 if (trigger_from_pin) val |= CfgChReg::TRGNSPI_BIT;
467 val |= (static_cast<uint32_t>(hit_raw & 0x7F) << CfgChReg::HIT_SHIFT);
468 val |= (static_cast<uint32_t>(hit_time_raw) << CfgChReg::HITT_SHIFT);
469 if (drive_mode == DriveMode::VDR) val |= CfgChReg::VDRNCDR_BIT;
470 if (side_mode == SideMode::HIGH_SIDE) val |= CfgChReg::HSNLS_BIT;
471 val |= (static_cast<uint32_t>(chop_freq) << CfgChReg::FREQ_CFG_SHIFT);
472 if (slew_rate_control_enabled) val |= CfgChReg::SRC_BIT;
473 if (open_load_detection_enabled) val |= CfgChReg::OL_EN_BIT;
474 if (plunger_movement_detection_enabled) val |= CfgChReg::DPM_EN_BIT;
475 if (hit_current_check_enabled) val |= CfgChReg::HHF_EN_BIT;
476 return val;
477 }

489 void fromRegister(uint32_t val, uint32_t full_scale_current_ma_param, bool master_clock_80khz_param) {
490 // Parse register fields
491 half_full_scale = (val & CfgChReg::HFS_BIT) != 0;
492 uint8_t hold_raw = static_cast<uint8_t>((val >> CfgChReg::HOLD_SHIFT) & 0x7F);
493 trigger_from_pin = (val & CfgChReg::TRGNSPI_BIT) != 0;
494 uint8_t hit_raw = static_cast<uint8_t>((val >> CfgChReg::HIT_SHIFT) & 0x7F);
495 uint8_t hit_time_raw = static_cast<uint8_t>((val >> CfgChReg::HITT_SHIFT) & 0xFF);
496 drive_mode = (val & CfgChReg::VDRNCDR_BIT) ? DriveMode::VDR : DriveMode::CDR;
497 side_mode = (val & CfgChReg::HSNLS_BIT) ? SideMode::HIGH_SIDE : SideMode::LOW_SIDE;
498 chop_freq = static_cast<ChopFreq>((val >> CfgChReg::FREQ_CFG_SHIFT) & 0x03);
499 slew_rate_control_enabled = (val & CfgChReg::SRC_BIT) != 0;
500 open_load_detection_enabled = (val & CfgChReg::OL_EN_BIT) != 0;
501 plunger_movement_detection_enabled = (val & CfgChReg::DPM_EN_BIT) != 0;
502 hit_current_check_enabled = (val & CfgChReg::HHF_EN_BIT) != 0;
503
504 // Store context
505 full_scale_current_ma = full_scale_current_ma_param;
506 master_clock_80khz = master_clock_80khz_param;
507
508 // Convert raw → user units
509 if (drive_mode == DriveMode::CDR) {
510 // CDR: raw → mA
511 if (full_scale_current_ma > 0) {
512 hit_setpoint = (static_cast<float>(hit_raw) / 127.0f) * static_cast<float>(full_scale_current_ma);
513 hold_setpoint = (static_cast<float>(hold_raw) / 127.0f) * static_cast<float>(full_scale_current_ma);
514 } else {
515 hit_setpoint = 0.0f;
516 hold_setpoint = 0.0f;
517 }
518 } else {
519 // VDR: raw → duty percent
520 hit_setpoint = (static_cast<float>(hit_raw) / 127.0f) * 100.0f;
521 hold_setpoint = (static_cast<float>(hold_raw) / 127.0f) * 100.0f;
522 }
523
524 // Convert hit_time raw → ms
525 if (hit_time_raw == 0) {
526 hit_time_ms = 0.0f;
527 } else if (hit_time_raw == 255) {
528 hit_time_ms = -1.0f; // Continuous (infinite)
529 } else {
530 uint32_t fchop_khz = getChopFreqKhz(master_clock_80khz, chop_freq);
531 float fchop_hz = static_cast<float>(fchop_khz) * 1000.0f;
532 hit_time_ms = (static_cast<float>(hit_time_raw) * 40.0f / fchop_hz) * 1000.0f;
533 }
534 }
