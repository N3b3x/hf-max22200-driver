187 void GpioSet(max22200::CtrlPin pin, max22200::GpioSignal signal) {
188 int gpio_pin = -1;
189 int level = 0;
190 switch (pin) {
191 case max22200::CtrlPin::ENABLE:
192 gpio_pin = config_.enable_pin;
193 level = (signal == max22200::GpioSignal::ACTIVE) ? 1 : 0; // Active-high
194 break;
195 case max22200::CtrlPin::CMD:
196 gpio_pin = config_.cmd_pin;
197 level = (signal == max22200::GpioSignal::ACTIVE) ? 1 : 0; // Active-high
198 break;
199 case max22200::CtrlPin::FAULT:
200 return; // FAULT is read-only
201 }
202 if (gpio_pin >= 0) {
203 gpio_set_level(static_cast<gpio_num_t>(gpio_pin), level);
204 }
205 }

217 bool GpioRead(max22200::CtrlPin pin, max22200::GpioSignal &signal) {
218 if (pin != max22200::CtrlPin::FAULT || config_.fault_pin < 0) {
219 return false;
220 }
221 int level = gpio_get_level(static_cast<gpio_num_t>(config_.fault_pin));
222 // FAULT is active-low: physical 0 = fault present (ACTIVE)
223 signal = (level == 0) ? max22200::GpioSignal::ACTIVE
224 : max22200::GpioSignal::INACTIVE;
225 return true;

238 bool initializeGPIO() {
239 // Configure output pins (ENABLE, CMD)
240 auto configure_output = [this](int16_t pin, const char *name, int initial) -> bool {
241 if (pin < 0) return true; // Not configured, skip
242 gpio_config_t cfg = {
243 .pin_bit_mask = (1ULL << pin),
244 .mode = GPIO_MODE_OUTPUT,
245 .pull_up_en = GPIO_PULLUP_DISABLE,
246 .pull_down_en = GPIO_PULLDOWN_DISABLE,
247 .intr_type = GPIO_INTR_DISABLE};
248 if (gpio_config(&cfg) != ESP_OK) {
249 ESP_LOGE(TAG, "Failed to configure %s pin (GPIO%d)", name, pin);
250 return false;
251 }
252 gpio_set_level(static_cast<gpio_num_t>(pin), initial);
253 ESP_LOGI(TAG, "%s pin (GPIO%d) initialized, level=%d", name, pin, initial);
254 return true;
255 };
256 if (!configure_output(config_.enable_pin, "ENABLE", 0)) return false;
257 if (!configure_output(config_.cmd_pin, "CMD", 1)) return false; // CMD HIGH = SPI mode
258
259 // Configure FAULT as input with pull-up (active-low, open-drain)
260 if (config_.fault_pin >= 0) {
261 gpio_config_t cfg = {
262 .pin_bit_mask = (1ULL << config_.fault_pin),
263 .mode = GPIO_MODE_INPUT,
264 .pull_up_en = GPIO_PULLUP_ENABLE,
265 .pull_down_en = GPIO_PULLDOWN_DISABLE,
266 .intr_type = GPIO_INTR_DISABLE};
267 if (gpio_config(&cfg) != ESP_OK) {
268 ESP_LOGE(TAG, "Failed to configure FAULT pin (GPIO%d)", config_.fault_pin);
269 return false;
270 }
271 ESP_LOGI(TAG, "FAULT pin (GPIO%d) initialized as input", config_.fault_pin);
272 }
273 return true;
274 }
