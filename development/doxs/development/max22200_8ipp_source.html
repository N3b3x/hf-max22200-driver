45template <typename SpiType>
46DriverStatus MAX22200<SpiType>::Initialize() {
47 if (initialized_) {
48 return DriverStatus::OK;
49 }
50
51 // Step 0: Initialize SPI interface
52 if (!spi_interface_.Initialize()) {
53 updateStatistics(false);
54 return DriverStatus::INITIALIZATION_ERROR;
55 }
56
57 // Configure SPI: Mode 0 (CPOL=0, CPHA=0), MSB first
58 if (!spi_interface_.Configure(MAX_SPI_FREQ_STANDALONE_, 0, true)) {
59 updateStatistics(false);
60 return DriverStatus::INITIALIZATION_ERROR;
61 }
62
63 // Step 1: ENABLE pin HIGH, power-up delay
64 spi_interface_.GpioSetActive(CtrlPin::ENABLE);
65 spi_interface_.DelayUs(500); // 0.5ms power-up delay
66
67 constexpr int kMaxInitRetries = 3;
68 StatusConfig status;
69 DriverStatus result;
70
71 for (int attempt = 0; attempt < kMaxInitRetries; ++attempt) {
72 // Step 2: Read STATUS register to clear UVM and deassert nFAULT
73 result = ReadStatus(status);
74 if (result != DriverStatus::OK) {
75 spi_interface_.GpioSetInactive(CtrlPin::ENABLE);
76 updateStatistics(false);
77 return result;
78 }
79 if (GetLastFaultByte() == StatusReg::FAULT_BYTE_COMER) {
80 continue; // Communication error — retry per datasheet Figure 6
81 }
82
83 // Step 3: Write STATUS register — set ACTIVE=1, all channels off,
84 // channel-pair modes independent, mask communication fault (default)
85 status.active = true;
86 status.channels_on_mask = 0x00; // All channels off initially
87 status.communication_error_masked = true; // Mask communication fault (reset default)
88
89 result = WriteStatus(status);
90 if (result != DriverStatus::OK) {
91 spi_interface_.GpioSetInactive(CtrlPin::ENABLE);
92 updateStatistics(false);
93 return result;
94 }
95 if (GetLastFaultByte() == StatusReg::FAULT_BYTE_COMER) {
96 continue; // Communication error — retry
97 }
98
99 // Cache the status for ONCH updates
100 cached_status_ = status;
101
102 // Step 4: Read STATUS again to verify UVM is cleared
103 result = ReadStatus(status);
104 if (result != DriverStatus::OK) {
105 spi_interface_.GpioSetInactive(CtrlPin::ENABLE);
106 updateStatistics(false);
107 return result;
108 }
109 if (GetLastFaultByte() == StatusReg::FAULT_BYTE_COMER) {
110 continue; // Communication error — retry
111 }
112
113 if (status.undervoltage) {
114 // UVM still set — device may not have full power
115 // Continue anyway but note the condition
116 }
117
118 cached_status_ = status;
119 initialized_ = true;
120 updateStatistics(true);
121 return DriverStatus::OK;
122 }
123
124 // COMER persisted after all retries
125 spi_interface_.GpioSetInactive(CtrlPin::ENABLE);
126 updateStatistics(false);
127 return DriverStatus::COMMUNICATION_ERROR;
128}

130template <typename SpiType>
131DriverStatus MAX22200<SpiType>::Deinitialize() {
132 if (!initialized_) {
133 return DriverStatus::OK;
134 }
135
136 // Disable all channels
137 DisableAllChannels();
138
139 // Set ACTIVE=0
140 cached_status_.active = false;
141 cached_status_.channels_on_mask = 0;
142 WriteStatus(cached_status_);
143
144 // ENABLE pin LOW
145 spi_interface_.GpioSetInactive(CtrlPin::ENABLE);
146
147 initialized_ = false;
148 updateStatistics(true);
149 return DriverStatus::OK;
150}

185template <typename SpiType>
186DriverStatus MAX22200<SpiType>::ConfigureChannel(uint8_t channel,
187 const ChannelConfig &config) {
188 if (!IsValidChannel(channel)) {
189 updateStatistics(false);
190 return DriverStatus::INVALID_PARAMETER;
191 }
192
193 // Validate: ensure context is set for proper conversion
194 if (config.drive_mode == DriveMode::CDR && config.full_scale_current_ma == 0) {
195 updateStatistics(false);
196 return DriverStatus::INVALID_PARAMETER; // IFS required for CDR
197 }
198
199 // Datasheet: SRC mode only for fCHOP < 50 kHz
200 if (config.slew_rate_control_enabled &&
201 getChopFreqKhz(cached_status_.master_clock_80khz, config.chop_freq) >= 50) {
202 updateStatistics(false);
203 return DriverStatus::INVALID_PARAMETER;
204 }
205
206 uint32_t reg_val = config.toRegister();
207 DriverStatus result = writeReg32(getChannelCfgBank(channel), reg_val);
208 updateStatistics(result == DriverStatus::OK);
209 return result;
210}

308template <typename SpiType>
309DriverStatus MAX22200<SpiType>::SetFullBridgeState(uint8_t pair_index,
310 FullBridgeState state) {
311 if (pair_index > 3) {
312 updateStatistics(false);
313 return DriverStatus::INVALID_PARAMETER;
314 }
315 const uint8_t shift = pair_index * 2;
316 const uint8_t pair_mask = 3u << shift;
317 uint8_t bits;
318 switch (state) {
319 case FullBridgeState::HiZ:
320 bits = 0;
321 break;
322 case FullBridgeState::Forward:
323 bits = 1u << shift; // ONCHx = 1, ONCHy = 0
324 break;
325 case FullBridgeState::Reverse:
326 bits = 2u << shift; // ONCHx = 0, ONCHy = 1
327 break;
328 case FullBridgeState::Brake:
329 bits = 3u << shift;
330 break;
331 default:
332 bits = 0;
333 break;
334 }
335 cached_status_.channels_on_mask = (cached_status_.channels_on_mask & ~pair_mask) | bits;
336 return SetChannelsOn(cached_status_.channels_on_mask);
337}

425template <typename SpiType>
426DriverStatus MAX22200<SpiType>::ConfigureDpm(float start_current_ma,
427 float dip_threshold_ma,
428 float debounce_ms) {
429 if (board_config_.full_scale_current_ma == 0) {
430 updateStatistics(false);
431 return DriverStatus::INVALID_PARAMETER;
432 }
433 DpmConfig config;
434 DriverStatus result = ReadDpmConfig(config);
435 if (result != DriverStatus::OK) {
436 return result;
437 }
438 // ISTART = plunger_movement_start_current × (IFS/127) => value = round(start_ma / IFS_ma * 127)
439 float ratio = start_current_ma / static_cast<float>(board_config_.full_scale_current_ma);
440 config.plunger_movement_start_current = static_cast<uint8_t>(ratio * 127.0f + 0.5f);
441 if (config.plunger_movement_start_current > 127) config.plunger_movement_start_current = 127;
442 ratio = dip_threshold_ma / static_cast<float>(board_config_.full_scale_current_ma);
443 uint32_t ipth = static_cast<uint32_t>(ratio * 127.0f + 0.5f);
444 config.plunger_movement_current_threshold = (ipth > 15) ? 15 : static_cast<uint8_t>(ipth);
445 // TDEB = plunger_movement_debounce_time / fCHOP; use default fCHOP = 25 kHz => value = debounce_ms * 25
446 uint32_t periods = static_cast<uint32_t>(debounce_ms * 25.0f + 0.5f);
447 config.plunger_movement_debounce_time = (periods > 15) ? 15 : static_cast<uint8_t>(periods);
448 return WriteDpmConfig(config);
449}

551// ============================================================================

561template <typename SpiType>
562DriverStatus MAX22200<SpiType>::writeCommandRegister(uint8_t bank,
563 bool is_write,
564 bool mode8) const {
565 uint8_t cmd_byte = CommandReg::build(bank, is_write, mode8);
566 uint8_t rx_byte = 0xFF;
567
568 // CMD pin HIGH for Command Register write
569 spi_interface_.GpioSetActive(CtrlPin::CMD);
570
571 // CS LOW → transfer 1 byte → CS HIGH
572 // Note: ESP-IDF handles CS automatically in the Transfer call
573 bool success = spi_interface_.Transfer(&cmd_byte, &rx_byte, 1);
574
575 // CMD pin LOW (prepare for data phase)
576 spi_interface_.GpioSetInactive(CtrlPin::CMD);
577
578 if (!success) {
579 return DriverStatus::COMMUNICATION_ERROR;
580 }
581
582 // Store the fault flags byte returned by the device
583 last_fault_byte_ = rx_byte;
584
585 return DriverStatus::OK;
586}

1015template <typename SpiType>
1016DriverStatus MAX22200<SpiType>::GetDutyLimits(bool master_clock_80khz, ChopFreq chop_freq,
1017 bool slew_rate_control_enabled, DutyLimits &limits) {
1018 // Compute fCHOP from FREQM and FREQ_CFG (datasheet Table 1)
1019 uint32_t fchop_khz;
1020 switch (chop_freq) {
1021 case ChopFreq::FMAIN_DIV4:
1022 fchop_khz = master_clock_80khz ? 20 : 25;
1023 break;
1024 case ChopFreq::FMAIN_DIV3:
1025 fchop_khz = master_clock_80khz ? 26 : 33; // 26.66 -> 26, 33.33 -> 33
1026 break;
1027 case ChopFreq::FMAIN_DIV2:
1028 fchop_khz = master_clock_80khz ? 40 : 50;
1029 break;
1030 case ChopFreq::FMAIN:
1031 fchop_khz = master_clock_80khz ? 80 : 100;
1032 break;
1033 default:
1034 limits = DutyLimits();
1035 return DriverStatus::INVALID_PARAMETER;
1036 }
1037
1038 // Datasheet Table 2: Min and Max Duty Cycle
1039 if (slew_rate_control_enabled) {
1040 // SRC = 1 (Slew-Rate Controlled)
1041 if (fchop_khz == 100 || fchop_khz == 80) {
1042 // FREQ_CFG = 11 with SRC=1: not valid per datasheet (SRC only for fCHOP < 50kHz)
1043 limits = DutyLimits();
1044 return DriverStatus::INVALID_PARAMETER;
1045 }
1046 // SRC=1: δMIN = 7%, δMAX = 93% for valid frequencies
1047 limits.min_percent = 7;
1048 limits.max_percent = 93;
1049 } else {
1050 // SRC = 0 (Fast Mode)
1051 if (fchop_khz == 100 || fchop_khz == 80) {
1052 // FREQ_CFG = 11: δMIN = 8%, δMAX = 92%
1053 limits.min_percent = 8;
1054 limits.max_percent = 92;
1055 } else {
1056 // Other FREQ_CFG: δMIN = 4%, δMAX = 96%
1057 limits.min_percent = 4;
1058 limits.max_percent = 96;
1059 }
1060 }
1061
1062 return DriverStatus::OK;
1063}
