35// Initialize the MAX22200 driver
36template <typename SpiType> DriverStatus MAX22200<SpiType>::Initialize() {
37 // Check if already initialized
38 if (initialized_) {
39 return DriverStatus::OK;
40 }
41
42 // Initialize SPI interface
43 if (!spi_interface_.Initialize()) {
44 updateStatistics(false);
45 return DriverStatus::INITIALIZATION_ERROR;
46 }
47
48 // Configure SPI for MAX22200
49 if (!spi_interface_.Configure(MAX_SPI_FREQ_STANDALONE_, 0, true)) {
50 updateStatistics(false);
51 return DriverStatus::INITIALIZATION_ERROR;
52 }
53
54 // Enable the device (ENABLE pin active-high). Per MAX22200 startup flowchart,
55 // a power-up delay is required before any SPI; the comm implementation provides it.
56 spi_interface_.GpioSetActive(CtrlPin::ENABLE);
57 spi_interface_.DelayUs(500); // POWER-UP 0.5 ms minimum before first SPI
58
59 // Reset the device
60 DriverStatus status = Reset();
61 if (status != DriverStatus::OK) {
62 spi_interface_.GpioSetInactive(CtrlPin::ENABLE);
63 updateStatistics(false);
64 return status;
65 }
66
67 // Configure global settings
68 GlobalConfig global_config;
69 global_config.diagnostic_enable = diagnostics_enabled_;
70 global_config.ics_enable = true;
71 global_config.daisy_chain_mode = false;
72 global_config.sleep_mode = false;
73 global_config.reset = false;
74
75 status = ConfigureGlobal(global_config);
76 if (status != DriverStatus::OK) {
77 spi_interface_.GpioSetInactive(CtrlPin::ENABLE);
78 updateStatistics(false);
79 return status;
80 }
81
82 // Clear any existing faults
83 ClearFaultStatus();
84
85 initialized_ = true;
86 updateStatistics(true);
87 return DriverStatus::OK;
88}

90// Deinitialize the MAX22200 driver. ENABLE is deasserted last so the device is
91// fully off after channels are disabled and sleep is set.
92template <typename SpiType> DriverStatus MAX22200<SpiType>::Deinitialize() {
93 if (!initialized_) {
94 return DriverStatus::OK;
95 }
96
97 // Safe shutdown: disable outputs, then sleep, then remove enable.
98 EnableAllChannels(false);
99 SetSleepMode(true);
100 spi_interface_.GpioSetInactive(CtrlPin::ENABLE);
101
102 initialized_ = false;
103 updateStatistics(true);
104 return DriverStatus::OK;
105}

245// Configure a specific channel
246template <typename SpiType>
247DriverStatus MAX22200<SpiType>::ConfigureChannel(uint8_t channel,
248 const ChannelConfig &config) {
249 if (!IsValidChannel(channel)) {
250 updateStatistics(false);
251 return DriverStatus::INVALID_PARAMETER;
252 }
253
254 // Validate current values
255 if (config.hit_current > CurrentRange::MAX_HIT_CURRENT ||
256 config.hold_current > CurrentRange::MAX_HOLD_CURRENT ||
257 config.hit_time > TimingRange::MAX_HIT_TIME) {
258 updateStatistics(false);
259 return DriverStatus::INVALID_PARAMETER;
260 }
261
262 // Channel config order per MAX22200: CFG (drive/bridge/parallel/polarity),
263 // then current (HIT/HOLD), then timing (hit time), then channel enable.
264 uint16_t config_value = buildChannelConfigValue(config);
265 DriverStatus status =
266 writeRegister(getChannelConfigReg(channel), config_value);
267 if (status != DriverStatus::OK) {
268 updateStatistics(false);
269 return status;
270 }
271
272 // Write current settings
273 uint16_t current_value = (config.hit_current << 8) | config.hold_current;
274 status = writeRegister(getChannelCurrentReg(channel), current_value);
275 if (status != DriverStatus::OK) {
276 updateStatistics(false);
277 return status;
278 }
279
280 // Write timing settings
281 status = writeRegister(getChannelTimingReg(channel), config.hit_time);
282 if (status != DriverStatus::OK) {
283 updateStatistics(false);
284 return status;
285 }
286
287 // Apply channel enable state to CHANNEL_ENABLE register (per-flowchart ONCHx)
288 status = EnableChannel(channel, config.enabled);
289 updateStatistics(status == DriverStatus::OK);
290 return status;
291}

299// Get configuration of a specific channel
300template <typename SpiType>
301DriverStatus MAX22200<SpiType>::GetChannelConfig(uint8_t channel,
302 ChannelConfig &config) const {
303 if (!IsValidChannel(channel)) {
304 updateStatistics(false);
305 return DriverStatus::INVALID_PARAMETER;
306 }
307
308 // Read channel configuration (mask to defined bits 0-3; reserved bits
309 // may read undefined on device or when not connected)
310 uint16_t config_value;
311 DriverStatus status =
312 readRegister(getChannelConfigReg(channel), config_value);
313 if (status != DriverStatus::OK) {
314 updateStatistics(false);
315 return status;
316 }
317 constexpr uint16_t CH_CONFIG_MASK = ChannelConfigBits::DRIVE_MODE_MASK |
318 ChannelConfigBits::BRIDGE_MODE_MASK |
319 ChannelConfigBits::PARALLEL_MASK |
320 ChannelConfigBits::POLARITY_MASK;
321 config = parseChannelConfigValue(config_value & CH_CONFIG_MASK);
322
323 // Read current settings
324 uint16_t current_value;
325 status = readRegister(getChannelCurrentReg(channel), current_value);
326 if (status == DriverStatus::OK) {
327 config.hit_current = (current_value >> 8) & 0x03FF;
328 config.hold_current = current_value & 0x03FF;
329 }
330
331 // Read timing settings
332 uint16_t timing_value;
333 status = readRegister(getChannelTimingReg(channel), timing_value);
334 if (status == DriverStatus::OK) {
335 config.hit_time = timing_value;
336 }
337
338 // Read enable state from channel enable register
339 uint16_t enable_value;
340 status = readRegister(Registers::CHANNEL_ENABLE, enable_value);
341 if (status == DriverStatus::OK) {
342 config.enabled = (enable_value & (1 << channel)) != 0;
343 }
344
345 updateStatistics(status == DriverStatus::OK);
346 return status;
347}

726// Write register
727template <typename SpiType>
728DriverStatus MAX22200<SpiType>::writeRegister(uint8_t reg,
729 uint16_t value) const {
730 uint8_t tx_data[3] = {reg, static_cast<uint8_t>(value >> 8),
731 static_cast<uint8_t>(value & 0xFF)};
732 uint8_t rx_data[3];
733
734 spi_interface_.SetChipSelect(false);
735 bool success = spi_interface_.Transfer(tx_data, rx_data, 3);
736 spi_interface_.SetChipSelect(true);
737
738 if (!success) {
739 return DriverStatus::COMMUNICATION_ERROR;
740 }
741
742 return DriverStatus::OK;
743}

745// Read register
746template <typename SpiType>
747DriverStatus MAX22200<SpiType>::readRegister(uint8_t reg,
748 uint16_t &value) const {
749 uint8_t tx_data[3] = {static_cast<uint8_t>(reg | 0x80), 0x00, 0x00};
750 uint8_t rx_data[3];
751
752 spi_interface_.SetChipSelect(false);
753 bool success = spi_interface_.Transfer(tx_data, rx_data, 3);
754 spi_interface_.SetChipSelect(true);
755
756 if (!success) {
757 return DriverStatus::COMMUNICATION_ERROR;
758 }
759
760 value = (static_cast<uint16_t>(rx_data[1]) << 8) | rx_data[2];
761 return DriverStatus::OK;
762}

764// Write register array
765template <typename SpiType>
766DriverStatus MAX22200<SpiType>::writeRegisterArray(uint8_t reg,
767 const uint8_t *data,
768 size_t length) const {
769 uint8_t *tx_data = new uint8_t[length + 1];
770 uint8_t *rx_data = new uint8_t[length + 1];
771
772 tx_data[0] = reg;
773 memcpy(&tx_data[1], data, length);
