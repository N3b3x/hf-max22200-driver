Build a Command Register value.

Build a Command Register valueConstructs the 8-bit Command Register byte from the register bank address, read/write mode, and transfer size.

Parameters
bankRegister bank address (0x00-0x0A, see RegBank namespace)
writetrue for write operation, false for read
mode8true for 8-bit MSB only, false for 32-bit full register
Returns
8-bit command register value ready for SPI transfer
// Write 32-bit STATUS register
uint8_t cmd = CommandReg::build(RegBank::STATUS, true, false);
// Result: 0x80 | (0x00 << 1) | 0x00 = 0x80
// Read 8-bit MSB of CFG_CH0 (fast HOLD current update)
uint8_t cmd = CommandReg::build(RegBank::CFG_CH0, false, true);
// Result: 0x00 | (0x01 << 1) | 0x01 = 0x03

// ============================================================================
// Command Register (8-bit, Write Only)
// ============================================================================
namespace CommandReg {
constexpr uint8_t RBW_POS = 7;
constexpr uint8_t RBW_READ = 0x00;
constexpr uint8_t RBW_WRITE = 0x80;
constexpr uint8_t A_BNK_POS = 1;
constexpr uint8_t A_BNK_MASK = 0x1E;
constexpr uint8_t MODE_8BIT = 0x01;
constexpr uint8_t MODE_32BIT = 0x00;
constexpr uint8_t build(uint8_t bank, bool write, bool mode8 = false) {
return static_cast<uint8_t>(
(write ? RBW_WRITE : RBW_READ) |
((bank & 0x0F) << A_BNK_POS) |
(mode8 ? MODE_8BIT : MODE_32BIT));
}
} // namespace CommandReg
